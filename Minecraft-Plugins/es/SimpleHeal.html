#   update-message: activar o desactivar el mensaje de actualizacion disponible al entrar al servidor
#   no-argument-message: activar o desactivar el mensaje del comando "/sh" en el caso de no tener ningun argumento
#   no-perm-message: activar o desactivar el mensaje para cuando usas un comando y no tienes permiso

#   heal-message: activar o desactivar el mensaje del comando "/heal"
#   pay-to-heal: activar o desactivar la opcion de pagar para curar (si se activa es necesario el plugin Vault y EssentialsX)
#   pay-to-heal-price: cantidad a pagar al usar "/heal" si la opcion (pay-to-heal) esta activa

#   wait-to-heal: activar o desactiva la opcion de esperar para curar
#   wait-to-heal-time: cantidad de tiempo (en segundos) que debes esperar para usar el comando "/heal"


Config:
  update-message: true
  no-argument-message: true
  no-perm-message: true
  heal-message: true
  pay-to-heal: false
  pay-to-heal-price: 1000
  wait-to-heal: true
  wait-to-heal-time: 300

Messages:
  heal:
    heal: '%plugin% &aHas sido curado'
    heal-others: '%plugin% &aHas curado a &6%target%'
    not-enought-money: '%plugin% &cNo tienes suficiente dinero para curarte'
    not-enought-money-others: '%plugin% &cNo tienes suficiente dinero para curar a &6%target%'
    wait-time-heal: '%plugin% &cDebes esperar &6%seconds-left% &csegundos para usar ese comando'
    wait-time-heal-others: '%plugin% &cDebes esperar &6%seconds-left% &csegundos para curar a &6%target%'
  help:
  - '&b<-----------------&aCOMANDOS&b----------------->'
  - '&e/sh help &7Muestra este mensaje'
  - '&e/sh version &7Version del plugin'
  - '&e/sh reload &7Recargar el plugin'
  - '&e/sh plugin &7Muestra el link de descarga del plugin'
  - '&e/heal &7Recuperar vida'
  - '&e/heal <player> &7Recuperar vida a un jugador'
  reload: '%plugin% &aEl plugin ha sido recargado'
  version: '%plugin% &aVersion &b%version%'
  command-no-argument: '%plugin% &fUsa &e/sh help &fpara ver la lista de comandos'
  no-perm: '%plugin% &cNo tienes permiso para ejecutar ese comando'
