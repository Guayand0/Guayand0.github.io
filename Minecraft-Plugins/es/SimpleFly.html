#   update-message: activar o desactivar el mensaje de actualizacion disponible al entrar al servidor
#   no-argument-message: activar o desactivar el mensaje del comando "/sf" en el caso de no tener ningun argumento
#   no-perm-message: activar o desactivar el mensaje para cuando usas un comando y no tienes permiso

#   fly-message: activar o desactivar el mensaje del comando "/fly"
#   pay-to-fly: activar o desactivar la opcion de pagar para volar (si se activa es necesario el plugin Vault y EssentialsX)
#   fly-price: cantidad a pagar al usar "/fly" si la opcion (pay-to-fly) esta activa


Config:
  update-message: true
  no-argument-message: true
  no-perm-message: true
  fly-message: true
  pay-to-fly: false
  fly-price: 1000

Messages:
  fly:
    enabled: '%plugin% &aFly ha sido activado'
    disabled: '%plugin% &cFly ha sido desactivado'
    enabled-others: '%plugin% &aHas activado el fly a &6%target%'
    disabled-others: '%plugin% &Has desactivado el fly a &6%target%'
    no-enought-money: '%plugin% &cNo tienes suficiente dinero para volar'
    no-enought-money-others: '%plugin% &cNo tienes suficiente dinero para hacer volar a &6%target%'
  help:
  - '&b<-----------------&aCOMANDOS&b----------------->'
  - '&e/sf help &7Muestra este mensaje'
  - '&e/sf version &7Version del plugin'
  - '&e/sf reload &7Recargar el plugin'
  - '&e/sf plugin &7Muestra el link de descarga del plugin'
  - '&e/fly &7Activa o desactiva fly'
  - '&e/fly <player> &7Activa o desactiva flyfly para otro jugador'
  reload: '%plugin% &aEl plugin ha sido recargado'
  version: '%plugin% &aVersion &b%version%'
  command-no-argument: '%plugin% &fUsa &e/sf help &fpara ver la lista de comandos'
  no-perm: '%plugin% &cNo tienes permiso para ejecutar ese comando'
  console-error: 'No puedes ejecutar ese comando desde la consola'
  update-checker: '%plugin% &bHay una nueva version &e(&f%latestversion%&e)&b. Descargala aqui: &7https://www.spigotmc.org/resources/111056/'
